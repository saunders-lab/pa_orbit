theme_figure <- function () {
  theme_classic( ) %+replace%
    theme(
      axis.line = element_line(color = 'black', size = 0.25),
      axis.ticks = element_line(color = 'black', size =0.25),
      axis.text = element_text(color = 'black', size=7),
      axis.title=element_text(color = 'black', size=7),
      strip.text = element_text(color = 'black', size = 7),
      strip.background = element_blank(),
      legend.background = element_blank(),
      legend.title=element_text(color = 'black',size=7),
      legend.text=element_text(color = 'black',size=7),
      legend.text.align=0,
      panel.spacing = unit(0,'cm'),
      plot.margin = margin(t=0.25, b = 0.25, l = 0.25, r = 0.25, unit = 'cm'),
      plot.title = element_text(hjust = 0.5, color = 'black', size = 7),
      plot.subtitle = element_text(hjust = 0.5, color = 'black', size = 7),
      plot.caption = element_text(color = 'black', size = 8)
    )
}

Fig. 3B - Phylogeny and genomes

df_genomes <- read_delim("../../data/low_throughput/clinical_genomes_fastANI.txt.visual", col_names = c('genome_1','genome_2','ANI','Q1','Q2','Q3','pos_1_start','pos_1_end','pos_2_start','pos_2_end') )

df_lens <- df_genomes %>% 
  group_by(genome_1) %>% 
  summarise(genome_len = max(pos_1_end))

df_orth <- df_genomes %>% 
  filter(genome_2 == 'Pseudomonas_aeruginosa_PAO1_107.fasta') %>% 
  mutate(frag_len = pos_1_end - pos_1_start) %>% 
  group_by(genome_1) %>% 
  summarise(orth_len = sum(frag_len))

left_join(df_orth, df_lens, by = 'genome_1') %>% 
  mutate(frac_orth = orth_len / genome_len) %>% 
  mutate(frac_diff = 1-frac_orth) %>% 
  mutate(length_diff = genome_len - orth_len)

Fig. 3C - Clinical Accuracy

df_clin <- read_csv("../../data/low_throughput/2025_12_23_clinical_acc.csv") %>% 
  pivot_longer(cols = -strain, names_to = 'locus', values_to = 'accuracy') %>% 
  mutate(strain = factor(strain))

df_clin
plot_clin <- df_clin %>% 
  mutate(locus = factor(locus, levels = c('HisA','MetX','ClpP'))) %>% 
  mutate(strain = factor(strain, levels = c('3301','3146','4041'))) %>% 
ggplot(aes(x = locus, y = accuracy / 100, fill = locus))+
  geom_col()+
  scale_y_continuous(label = scales::label_percent())+
  facet_wrap(~strain)+
    scale_fill_manual(values = c('#738FC1', '#D56C55','#ab85ac'))+
  labs(x = NULL, y = 'Accuracy') + guides(fill = 'none')

df_mic <- read_csv("../../data/low_throughput/2026_02_06_PA_MIC_mexAC.csv")

df_mic %>% 
  mutate(locus = factor(locus, levels = c('WT','mexA','mexC'))) %>% 
  mutate(strain = factor(strain, levels = c('PAO1','PA14','3301'))) %>%
  mutate(drug = factor(drug, levels = c('Levo','Cef','PipT'))) %>% 
  filter(drug != 'PipT') %>% 
ggplot(aes(x = strain, y = locus, fill = MIC)) + geom_tile() + facet_wrap(~drug) +
